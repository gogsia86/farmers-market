## 5. Image Optimization Script

Create `optimize-images.js`:
```javascript
const sharp = require('sharp');
const fs = require('fs').promises;
const path = require('path');

class ImageOptimizer {
    constructor() {
        this.sizes = {
            hero: { width: 1920, height: 1080 },
            product: { width: 800, height: 600 },
            vendor: { width: 600, height: 600 },
            icon: { width: 100, height: 100 }
        };
    }

    async optimizeAll() {
        const directories = [
            'heroes',
            'products/vegetables',
            'products/fruits', 
            'products/dairy',
            'products/bakery',
            'vendors',
            'icons'
        ];

        for (const dir of directories) {
            await this.optimizeDirectory(dir);
        }
    }

    async optimizeDirectory(directory) {
        const files = await fs.readdir(`./original-images/${directory}`);
        
        for (const file of files) {
            if (file.endsWith('.jpg') || file.endsWith('.png')) {
                await this.optimizeImage(path.join(directory, file));
            }
        }
    }

    async optimizeImage(filePath) {
        const size = this.getSizeForDirectory(path.dirname(filePath));
        
        try {
            await sharp(`./original-images/${filePath}`)
                .resize(size.width, size.height, {
                    fit: 'cover',
                    position: 'center'
                })
                .jpeg({ quality: 80 })
                .toFile(`./public/images/${filePath.replace('.png', '.jpg')}`);
            
            console.log(`Optimized: ${filePath}`);
        } catch (error) {
            console.error(`Error optimizing ${filePath}:`, error);
        }
    }

    getSizeForDirectory(directory) {
        if (directory.includes('hero')) return this.sizes.hero;
        if (directory.includes('products')) return this.sizes.product;
        if (directory.includes('vendor')) return this.sizes.vendor;
        if (directory.includes('icon')) return this.sizes.icon;
        return this.sizes.product;
    }
}

// Run optimization
const optimizer = new ImageOptimizer();
optimizer.optimizeAll();
```

## 7. Image Placeholder Service

For development, add this to your `server.js`:
```javascript
// Image placeholder route for development
app.get('/images/placeholder/:width/:height', (req, res) => {
    const { width, height } = req.params;
    const text = `${width}x${height}`;
    
    const svg = `
        <svg width="${width}" height="${height}" xmlns="http://www.w3.org/2000/svg">
            <rect width="100%" height="100%" fill="#4a7c59"/>
            <text x="50%" y="50%" font-family="Arial" font-size="24" fill="white" text-anchor="middle" dy=".3em">${text}</text>
        </svg>
    `;
    
    res.set('Content-Type', 'image/svg+xml');
    res.send(svg);
});
```
