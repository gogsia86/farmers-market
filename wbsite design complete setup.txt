// Main application JavaScript
class FarmersMarketApp {
    constructor() {
        this.currentPage = 'home';
        this.cart = JSON.parse(localStorage.getItem('cart')) || [];
        this.init();
    }

    init() {
        this.setupEventListeners();
        this.loadPage(window.location.pathname);
        this.updateCartCount();
    }

    setupEventListeners() {
        // Navigation
        document.addEventListener('click', (e) => {
            if (e.target.matches('.nav-link') || e.target.closest('.nav-link')) {
                e.preventDefault();
                const link = e.target.closest('.nav-link');
                const href = link.getAttribute('href');
                this.loadPage(href);
            }

            if (e.target.matches('.hamburger') || e.target.closest('.hamburger')) {
                this.toggleMobileMenu();
            }
        });

        // Close mobile menu when clicking outside
        document.addEventListener('click', (e) => {
            if (!e.target.closest('.nav-container') && document.querySelector('.nav-menu').classList.contains('active')) {
                this.toggleMobileMenu();
            }
        });

        // Handle browser back/forward
        window.addEventListener('popstate', (e) => {
            this.loadPage(window.location.pathname, false);
        });
    }

    async loadPage(path, pushState = true) {
        try {
            // Update active nav link
            this.updateActiveNavLink(path);

            // Show loading state
            document.getElementById('main-content').innerHTML = '<div class="loading" style="text-align: center; padding: 2rem;"></div>';

            let content = '';
            
            switch (path) {
                case '/':
                case '/home':
                    content = await this.loadHomePage();
                    this.currentPage = 'home';
                    break;
                case '/products':
                    content = await this.loadProductsPage();
                    this.currentPage = 'products';
                    break;
                case '/vendors':
                    content = await this.loadVendorsPage();
                    this.currentPage = 'vendors';
                    break;
                case '/about':
                    content = await this.loadAboutPage();
                    this.currentPage = 'about';
                    break;
                case '/cart':
                    content = await this.loadCartPage();
                    this.currentPage = 'cart';
                    break;
                default:
                    content = await this.loadHomePage();
            }

            document.getElementById('main-content').innerHTML = content;
            
            if (pushState) {
                window.history.pushState({}, '', path);
            }

            // Update cart count
            this.updateCartCount();

            // Re-initialize page-specific event listeners
            this.initializePageEvents();

        } catch (error) {
            console.error('Error loading page:', error);
            document.getElementById('main-content').innerHTML = this.getErrorPage();
        }
    }

            }
        });
    }

// Cart functionality
FarmersMarketApp.prototype.addToCart = function(productId) {
    this.fetchProducts().then(products => {
        const product = products.find(p => p.id === productId);
        if (!product) return;

        const existingItem = this.cart.find(item => item.id === productId);
        
        if (existingItem) {
            existingItem.quantity += 1;
        } else {
            this.cart.push({
                ...product,
                quantity: 1
            });
        }

        this.saveCart();
        this.updateCartCount();
        this.showAlert(`${product.name} added to cart!`, 'success');
    });
};

FarmersMarketApp.prototype.removeFromCart = function(productId) {
    this.cart = this.cart.filter(item => item.id !== productId);
    this.saveCart();
    this.updateCartCount();
    
    if (this.currentPage === 'cart') {
        this.loadCartPage().then(content => {
            document.getElementById('main-content').innerHTML = content;
        });
    }
};

FarmersMarketApp.prototype.updateQuantity = function(productId, newQuantity) {
    if (newQuantity < 1) {
        this.removeFromCart(productId);
        return;
    }

    const item = this.cart.find(item => item.id === productId);
    if (item) {
        item.quantity = newQuantity;
        this.saveCart();
        this.updateCartCount();
        
        if (this.currentPage === 'cart') {
            this.loadCartPage().then(content => {
                document.getElementById('main-content').innerHTML = content;
            });
        }
    }
};
